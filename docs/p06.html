  )

creating function with if() function.

grade <- function (x) {
  if (x>84){
    "Good"
  } else if (x>75){
    "So So"
  } else {
    "try again"
  }
}
grade(60)
## [1] "try again"

If () is for scalar. Ifelse () is for vector.

option <- function(x) {
  if (x == "a") {
    "awsome"
  } else if (x == "b") {
    "best" 
   } else {
    stop("Invalid `x` value")
  }
}

option2 <- function(x) {
  switch(x,
    a = "awsome",
    b = "best",

  )
